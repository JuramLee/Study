**********************
   eslint 설치 정리  
**********************

	1. npm i -D eslint eslint-plugin-prettier eslint-plugin-unused-imports eslint-plugin-react prettier babel-eslint @babel/eslint-parser
	2. src 바깥에다가 .prettierrc.js랑 .eslintrc.js 파일 생성
	3. 확장프로그램 ESLint 설치
	4. 설정 들어가서 오른쪽 끝에 파일모양 아이콘 눌러서 settings.json 열기
	5.   "editor.defaultFormatter": "esbenp.prettier-vscode", 
		"editor.codeActionsOnSave": {
			"source.fixAll.eslint": true
		},
		"editor.formatOnSave": true,
		"javascript.updateImportsOnFileMove.enabled": "always", 작성
	6. .eslintrc.js로 가서 아래 복붙
			--------------------------------------------------------
		module.exports = {
		env: {
		browser: true,
			es6: true,
		},
		parserOptions: {
			ecmaFeatures: {
			modules: true,
			jsx: true,
			},
			ecmaVersion: 2020,
			sourceType: 'module',
			babel: '@babel/eslint-parser',
		},
		plugins: ['react', 'unused-imports'],
		rules: {
			'react/jsx-uses-vars': 'error',
		'react/jsx-uses-react': 'error',
			'space-before-function-paren': [
				'error',
				{
				anonymous: 'ignore',
				named: 'never',
				asyncArrow: 'always',
				},
			],
			'unused-imports/no-unused-imports-ts': ['error'],
		},
		}
			--------------------------------------------------------
	7. .prettierrc.js로 가서 아래 복붙
		--------------------------------------------------------
			module.exports = {
				endOfLine: 'lf',
				semi: true,
				singleQuote: true,
				jsxSingleQuote: false,
				tabWidth: 2,
				useTabs: true,
				trailingComma: 'all',
				arrowParens: 'avoid',
			};
		--------------------------------------------------------
